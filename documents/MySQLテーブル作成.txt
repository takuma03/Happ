
■mysqlテーブル作成SQL
CREATE TABLE task
(
task_id INT(11) NOT NULL AUTO_INCREMENT PRIMARY KEY,
category VARCHAR(255),
task_name VARCHAR(255),
memo VARCHAR(255),
start_time datetime,
stop_time datetime,
update_time datetime
);
